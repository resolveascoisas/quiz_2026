function gerarTabelaEscolas(){ const tbody = document.querySelector( '#tabelaEscolas tbody' ) tbody.innerHTML = '' const agrupado = {} dadosGerais.forEach(item=>{ if(!agrupado[item.escola]){ agrupado[item.escola] = { somaPontos:0, totalSemanas:0, somaPresenca:0 } } agrupado[item.escola].somaPontos += item.pontos agrupado[item.escola].totalSemanas += 1 agrupado[item.escola].somaPresenca += item.presenca }) const ranking = Object.entries(agrupado) .map(([escola,dados])=>{ const mediaFinal = dados.somaPontos / dados.totalSemanas const mediaPresenca = dados.somaPresenca / dados.totalSemanas return { escola, media:mediaFinal, presenca:mediaPresenca, participacoes:dados.totalSemanas } }) .sort((a,b)=>{ // 1º critério = média if(b.media !== a.media){ return b.media - a.media } // 2º critério = frequência return b.presenca - a.presenca }) ranking.forEach((item,index)=>{ tbody.innerHTML += ` ${index + 1} ${item.escola} ${item.media.toFixed(2)} ${item.presenca.toFixed(1)}% ${item.participacoes} ` }) }
